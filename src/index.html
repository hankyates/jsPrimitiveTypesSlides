MDN 

        var Danny = {
          firstName: 'danny',
          lastName: 'yates',
          age: '24',
          fullName: function() {
            return this.firstname + ' ' + this.lastName;
          }
          
        };
        
        Danny.fullName();
        // 'danny yates'
        Danny.firstName = 'dan';
        Danny.fullName();
        // 'dan yates'
        Danny.firstName;
        // 'dan'
